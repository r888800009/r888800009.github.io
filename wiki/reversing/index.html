找出lib call

ltrace ./a.out

找出system call

strace ./a.out

顯示dynamic relocation資料

objdump -R a.out

顯示符號表

nm ./a.out

顯示區段

readelf -x .text ./a.out
objdump -s -j .text ./a.out

反組譯

objdump -M intel -d ./a.out | less
